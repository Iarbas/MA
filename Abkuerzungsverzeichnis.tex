  \acro{IMU}{Inertial Measurement Unit}{ (Inertialsensoren)}\\

  \acro{dBm}{Dezibel Milliwatt}{}\\

  \acro{LOS}{Line Of Sight}{ (Sichtverbindung)}\\
  \acro{NLOS}{Non Line Of Sight}{Nicht-Sichtverbindung}\\
  \acro{PSO}{Particel-Swarm-Opimization}{ (Partikel-Schwarm-Optimierung)}\\
\end{acronym}
